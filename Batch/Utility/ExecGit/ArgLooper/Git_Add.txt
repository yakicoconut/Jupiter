# # コマンドループ_Git_Add # #
# 概要
#   Gitファイル追加、コミットコマンド実行
# 使用方法
#   ・「#command:」を使用して管理ディレクトリに
#     カレントディレクトリを変更する前提
# 引数
#   add   :ファイル追加
#   commit:指定したファイルをコミットする
#          複数を対象とする場合、半角スペースで区切る
#          オプション
#            -m:コメント入力
#               (例:git commit -m "コメント" "ファイル1" "ファイル2"
# サイト
#
# オプションテンプレ
#   #command:
#   #option1:
#   #option2:

# カレントディレクトリ変更
#command:cd /d  対象ディレクトリ


# addオプション設定
#option1:add
# ファイル追加
にほへ.txt
ちりぬ.txt


# コミットオプション設定
#option1:commit -m
# コメント、ファイル指定コミット
"コメント1" "いろは.txt" "にほへ.txt"
